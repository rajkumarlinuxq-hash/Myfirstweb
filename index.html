// Dynamic 3D Jungle Scroll Matrix Engine const jCanvas = document.getElementById('jungleCinemaCanvas'); const jCtx = jCanvas.getContext('2d'); let scrollFraction = 0; function initJungleCinemaEngine() { jCanvas.width = window.innerWidth; jCanvas.height = window.innerHeight; } window.addEventListener('resize', () => { if(jCanvas.style.display === 'block') initJungleCinemaEngine(); }); window.addEventListener('scroll', () => { const scrollTop = window.scrollY; const docHeight = document.documentElement.scrollHeight - window.innerHeight; if(docHeight <= 0) return; scrollFraction = scrollTop / docHeight; if(document.getElementById('panel-about').classList.contains('active-panel')) { render3DJungleProductMatrix(scrollFraction); } }); // Ye function bilkul YouTube Shorts jaisa smooth scroll rotation effect dega (Jungle elements ke sath) function render3DJungleProductMatrix(fraction) { jCtx.clearRect(0, 0, jCanvas.width, jCanvas.height); // 1. Dynamic Background Color (Deep Forest Twilight to Bright Emerald-Green Glow on scroll) let r = Math.floor(3 + (fraction * 15)); let g = Math.floor(7 + (fraction * 45)); let b = Math.floor(18 + (fraction * 20)); jCtx.fillStyle = `rgb(${r}, ${g}, ${b})`; jCtx.fillRect(0, 0, jCanvas.width, jCanvas.height); const centerX = jCanvas.width / 2; const centerY = jCanvas.height / 2; // 2. Cinematic Light Rays (Angles rotate smoothly based on scroll position) jCtx.save(); jCtx.globalAlpha = 0.2 + (Math.sin(fraction * Math.PI) * 0.3); let rayAngle = fraction * Math.PI * 2; // Smooth rotation angle let lightGrad = jCtx.createRadialGradient(centerX, centerY, 10, centerX, centerY, jCanvas.width * 0.8); lightGrad.addColorStop(0, '#00ffcc'); lightGrad.addColorStop(0.5, 'rgba(157, 78, 221, 0.2)'); lightGrad.addColorStop(1, 'transparent'); jCtx.fillStyle = lightGrad; jCtx.translate(centerX, centerY); jCtx.rotate(rayAngle); jCtx.fillRect(-jCanvas.width, -jCanvas.height, jCanvas.width * 2, jCanvas.height * 2); jCtx.restore(); // 3. 3D Jungle Tree Matrix Layers (Moves closer & rotates into view on scroll) // Ye mimic karta hai video ke bottle components ke smooth transitions ko! let totalLayers = 12; for (let i = 0; i < totalLayers; i++) { // Depth simulation using scale and alpha let layerOffset = (i / totalLayers); let dynamicProgress = (fraction + layerOffset) % 1.0; let scale = Math.pow(dynamicProgress, 2) * 4; // Perspective 3D zoom out/in let alpha = Math.sin(dynamicProgress * Math.PI); if (scale < 0.1) continue; jCtx.save(); jCtx.translate(centerX, centerY); // Circular product/environment rotation like the DRIP website animation let layerRotation = (fraction * Math.PI * 1.5) + (i * 0.5); jCtx.rotate(layerRotation); jCtx.scale(scale, scale); jCtx.globalAlpha = alpha * 0.4; // Tree branch details rendering as premium vector matrix nodes jCtx.fillStyle = (i % 2 === 0) ? "rgba(0, 255, 204, 0.7)" : "rgba(157, 78, 221, 0.7)"; // Drawing custom 3D environmental abstract patterns/leaves jCtx.beginPath(); jCtx.moveTo(0, -50); jCtx.lineTo(30, 20); jCtx.lineTo(-30, 20); jCtx.closePath(); jCtx.fill(); // Adding glowing bio-luminescent jungle dots jCtx.fillStyle = "#ffffff"; jCtx.beginPath(); jCtx.arc(40, -20, 3 / scale, 0, Math.PI * 2); jCtx.fill(); jCtx.restore(); } // 4. Centerpiece Glowing 3D AI Matrix Tree (Zooms in and transforms parts dynamically) jCtx.save(); jCtx.translate(centerX, centerY); // Zoom and Rotate logic matching the scroll intensity of the laptop touchpad track let mainScale = 1.0 + (fraction * 0.8); let mainRotation = fraction * Math.PI * 0.5; // Smooth 90-degree showcase turn jCtx.scale(mainScale, mainScale); jCtx.rotate(mainRotation); // Draw Core Trunk (Glow effect) jCtx.shadowBlur = 30; jCtx.shadowColor = "#00ffcc"; jCtx.fillStyle = "rgba(13, 21, 39, 0.95)"; jCtx.strokeStyle = "#00ffcc"; jCtx.lineWidth = 4; // Interactive central block drawing jCtx.beginPath(); jCtx.rect(-40, -100, 80, 200); jCtx.fill(); jCtx.stroke(); // 5. Part Detachment Animation (Like the bottle cap opening in the video) // Jab scroll 60% se upar jayega, tab upper canopy alag hokar vertical fly karegi let canopyYOffset = 0; if (fraction > 0.5) { canopyYOffset = -(fraction - 0.5) * 140; // Moves upward smoothly } jCtx.fillStyle = "rgba(0, 210, 255, 0.8)"; jCtx.beginPath(); jCtx.arc(0, -100 + canopyYOffset, 70, 0, Math.PI, true); jCtx.fill(); jCtx.restore(); }
